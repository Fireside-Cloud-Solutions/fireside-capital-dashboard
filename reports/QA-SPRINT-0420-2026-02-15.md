/* FC-UIUX-003: Table Horizontal Overflow Fix (P1 HIGH) */

/* Ensure table containers allow horizontal scrolling */
.table-card,
.table-responsive {
  overflow-x: auto;
  -webkit-overflow-scrolling: touch;
  width: 100%;
  display: block;
}

}
