```javascript
// Calculate totals
let totalValue = 0;
let totalMonthlyContribution = 0;
let weightedReturn = 0;

investments.forEach(inv => {
  totalValue += parseFloat(inv.value) || 0;
  totalMonthlyContribution += parseFloat(inv.monthlyContribution) || 0;
  
  // Weight return by portfolio value
  if (inv.value > 0 && inv.annualReturn > 0) {
    weightedReturn += inv.annualReturn * inv.value;
  }
});

// Weighted average return
const avgReturn = totalValue > 0 ? (weightedReturn / totalValue) : 0;
```
